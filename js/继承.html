javascript的继承有原型链继承，借用构造函数继承

例子1-原型链继承

function A(){
	this.name = "aaname";
}
A.prototype.getName = function(){
	return this.name;
}
function B(){
	this.name = "bbname";
}
B.prototype = new A();
var bjcone = new B();
console.log(bjcone.getName());

例子2-借用构造函数继承

function C(){
	this.colors = ["red","green","blue","white","black"];
}
function D(){
	C.call(this);
}
var bone = new D();
bone.colors.push("yellow");
console.log(bone);
var btwo = new D();
console.log(btwo);
